SELECT nome, saldo
FROM clientes
WHERE saldo > (
    SELECT AVG(saldo)
    FROM clientes
);

SELECT nome
FROM clientes
WHERE id IN (
    SELECT cliente_id
    FROM pedidos
);

SELECT nome
FROM clientes
WHERE id NOT IN (
    SELECT cliente_id
    FROM pedidos
);

SELECT AVG(total)
FROM (
    SELECT cliente_id,
    SUM(valor) AS total
    FROM pedidos
    GROUP BY cliente_id
) AS totais;

SELECT nome
FROM clientes
WHERE id = (
    SELECT cliente_id
    FROM pedidos
    GROUP BY cliente_id
    ORDER BY SUM(valor) DESC
    LIMIT 1
);

SELECT nome
FROM clientes c
WHERE EXISTS (
    SELECT 1
    FROM pedidos p
    WHERE p.cliente_id = c.id
);

SELECT cidade,
AVG(total) AS media_cidade
FROM (
    SELECT c.cidade,
    SUM(p.valor) AS total
    FROM clientes c
    INNER JOIN pedidos p
    ON c.id = p.cliente_id
    GROUP BY c.nome, c.cidade
) AS vendas_clientes
GROUP BY cidade;

SELECT AVG(media)
FROM (
    SELECT cidade,
    AVG(saldo) AS media
    FROM clientes
    GROUP BY cidade
) AS medias_cidade;

select cidade, max(maximo), min(minimo)
from (
	select c2.cidade as cidade, max(c2.saldo) as maximo, min(c2.saldo) as minimo
	from clientes c2
	group by cidade
) as tabela_fantasma
group by cidade;

select avg(qtd)
from(
select p2.cliente_id, count(*) as qtd
from pedidos p2
group by p2.cliente_id
) as tabela_fantasma;

select c.nome, c.saldo, tabela_fantasma.total_gasto 
from (
	select p.cliente_id, sum(p.valor) as total_gasto
	from pedidos p
	group by p.cliente_id
) as tabela_fantasma
inner join clientes c
on c.id = tabela_fantasma.cliente_id
where c.saldo > 5000;
